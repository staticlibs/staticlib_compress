70  lzma_sink(Sink&& sink) :
71  sink(std::move(sink)),
72  strm([] {
73  lzma_stream* stream = static_cast<lzma_stream*> (std::malloc(sizeof(lzma_stream)));
74  if (nullptr == stream) throw compress_exception(TRACEMSG(
75  "Error creating lzma stream: 'malloc' failed"));
76  *stream = LZMA_STREAM_INIT;
77  auto err = ::lzma_easy_encoder(stream, compression_level, LZMA_CHECK_CRC64);
78  if (LZMA_OK != err) throw compress_exception(TRACEMSG(
79  "Error initializing LZMA stream, code: [" + sl::support::to_string(err) + "]"));
80  return stream;
81  }()) { }

83  ~lzma_sink() STATICLIB_NOEXCEPT {
84  if (nullptr == strm) return;
85  auto deferred = sl::support::defer([this]() STATICLIB_NOEXCEPT {
86  ::lzma_end(this->strm);
87  ::free(this->strm);
88  });
89  // finish encoding
90  strm->next_in = nullptr;
91  strm->avail_in = 0;
92  strm->next_out = reinterpret_cast<uint8_t*>(buf.data());
93  strm->avail_out = buf.size();
94  // call code
95  bool coding = true;
96  while (coding) {
97  auto err = ::lzma_code(strm, LZMA_FINISH);
98  switch (err) {
99  case LZMA_OK:
100  sl::io::write_all(sink, {buf.data(), buf.size() - strm->avail_out});
101  strm->next_out = reinterpret_cast<uint8_t*>(buf.data());
102  strm->avail_out = buf.size();
103  // not finished
104  break;
105  case LZMA_STREAM_END:
106  sl::io::write_all(sink, {buf.data(), buf.size() - strm->avail_out});
107  // finished
108  coding = false;
109  break;
110  default:
111  // finish
112  coding = false;
113  }
114  }
115  }

164  std::streamsize write(sl::io::span<const char> span) {
165  // prepare lzma stream
166  strm->next_in = reinterpret_cast<const uint8_t*>(span.data());
167  strm->avail_in = static_cast<size_t> (span.size());
168  strm->next_out = reinterpret_cast<uint8_t*>(buf.data());
169  strm->avail_out = buf.size();
170  // call code
171  while (strm->avail_in > 0) {
172  auto err = ::lzma_code(strm, LZMA_RUN);
173  switch (err) {
174  case LZMA_OK:
175  sl::io::write_all(sink, {buf.data(), buf.size() - strm->avail_out});
176  strm->next_out = reinterpret_cast<uint8_t*>(buf.data());
177  strm->avail_out = buf.size();
178  break;
179  default: throw compress_exception(TRACEMSG(
180  "LZMA error code: [" + sl::support::to_string(err) + "]"));
181  }
182  }
183  return span.size_signed();
184  }

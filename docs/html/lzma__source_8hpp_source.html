154  std::streamsize read(sl::io::span<char> span) {
155  if (!exhausted) {
156  // fill buffer if empty
157  if (0 == avail) {
158  avail = sl::io::read_all(src, {buf.data(), buf.size()});
159  pos = 0;
160  }
161  // prepare zlib stream
162  strm->next_in = reinterpret_cast<uint8_t*> (buf.data() + pos);
163  strm->avail_in = avail;
164  strm->next_out = reinterpret_cast<uint8_t*> (span.data());
165  strm->avail_out = static_cast<size_t> (span.size());
166  // call inflate
167  auto err = ::lzma_code(strm, LZMA_RUN);
168  if (LZMA_OK == err || LZMA_STREAM_END == err) {
169  std::streamsize read = avail - strm->avail_in;
170  std::streamsize written = span.size_signed() - strm->avail_out;
171  size_t uread = static_cast<size_t> (read);
172  pos += uread;
173  avail -= uread;
174  if (written > 0 || LZMA_STREAM_END != err) {
175  return written;
176  }
177  exhausted = true;
178  return std::char_traits<char>::eof();
179  } else throw compress_exception(TRACEMSG(
180  + "LZMA error, code: [" + sl::support::to_string(err) + "]"));
181  } else {
182  return std::char_traits<char>::eof();
183  }
184  }

277  std::streamsize write(sl::io::span<const char> span) {
278  if (nullptr != entry_deflater.get()) {
279  size_t count_before = entry_deflater->get_count();
280  entry_deflater->write(span);
281  size_t count = entry_deflater->get_count() - count_before;
282  this->entry_crc = ::crc32(entry_crc, reinterpret_cast<const Bytef*>(span.data()), static_cast<uInt>(count));
283  return static_cast<std::streamsize>(count);
284  } else {
285  throw compress_exception(TRACEMSG("Invalid ZIP sink state: add ZIP entry before writing the data"));

303  void add_entry(const std::string& filename) {
304  if (filename.empty()) throw compress_exception(TRACEMSG("Invalid empty entry name specified"));
305  if (cd_written) throw new compress_exception(TRACEMSG(
306  "Invalid entry add attempt for finalized ZIP stream"));
307  if (headers.size() > 0) {
308  write_entry_data_descriptor();
309  }
310  // add new entry
311  headers.emplace_back(std::string(filename.data(), filename.length()), static_cast<uint16_t>(method));
312  headers.back().write_local_file_header(sink, static_cast<uint32_t>(sink.get_count()));
313  entry_deflater.reset(new sl::io::counting_sink<deflate_sink<entry_counter_ref_type>>(make_deflate_sink(entry_counter)));
314  entry_crc = ::crc32(0L, Z_NULL, 0);
315  }
316 
322  void finalize() {
323  if (!cd_written && headers.size() > 0) {
324  write_entry_data_descriptor();
325  uint32_t cd_offset = static_cast<uint32_t>(sink.get_count());
326  for (detail::Header& he : headers) {
327  he.write_cd_file_header(sink);
328  }
329  size_t cd_len = sink.get_count() - cd_offset;
330  detail::write_eocd(sink, static_cast<uint16_t>(headers.size()), cd_offset, static_cast<uint32_t>(cd_len));
331  cd_written = true;
332  }
333  }

335 private:
336  void write_entry_data_descriptor() {
337  // get size and close current entry
338  uint32_t uncompressed_size = static_cast<uint32_t>(entry_deflater->get_count());
339  entry_deflater.reset(nullptr);
340  // get size and reset counter
341  uint32_t compressed_size = static_cast<uint32_t>(entry_counter.get_count());
342  entry_counter = sl::io::make_counting_sink(sink);
343  headers.back().write_data_descriptor(sink, compressed_size, uncompressed_size, entry_crc);
344  }

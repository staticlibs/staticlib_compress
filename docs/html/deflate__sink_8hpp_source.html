78  void operator()(z_stream* strm) {
79  // finish encoding
80  strm->next_in = nullptr;
81  strm->avail_in = 0;
82  strm->next_out = reinterpret_cast<unsigned char*> (buf);
83  strm->avail_out = static_cast<uInt> (buf_size);
84  // call deflate
85  for(;;) {
86  auto err = ::deflate(strm, Z_FINISH);
87  switch (err) {
88  case Z_OK:
89  staticlib::io::write_all(*sink, buf, buf_size - strm->avail_out);
90  strm->next_out = reinterpret_cast<unsigned char*> (buf);
91  strm->avail_out = static_cast<uInt> (buf_size);
92  break;
93  case Z_STREAM_END:
94  staticlib::io::write_all(*sink, buf, buf_size - strm->avail_out);
95  // fall through
96  default:
97  // cannot report any error safely - we are in destructor
98  goto end;
99  }
100  }
101  end:
102  ::deflateEnd(strm);
103  delete strm;
104  }
105 };

183  std::streamsize write(const char* buffer, std::streamsize len_in) {
184  // prepare zlib stream
185  strm->next_in = reinterpret_cast<const unsigned char*> (buffer);
186  strm->avail_in = static_cast<uInt> (len_in);
187  strm->next_out = reinterpret_cast<unsigned char*> (buf.data());
188  strm->avail_out = static_cast<uInt> (buf.size());
189  // call deflate
190  while(strm->avail_in > 0) {
191  auto err = ::deflate(strm.get(), Z_NO_FLUSH);
192  switch (err) {
193  case Z_OK:
194  staticlib::io::write_all(sink, buf.data(), buf.size() - strm->avail_out);
195  strm->next_out = reinterpret_cast<unsigned char*> (buf.data());
196  strm->avail_out = static_cast<uInt> (buf.size());
197  break;
198  default: throw CompressException(TRACEMSG(std::string() +
199  "Deflate error: [" + ::zError(err) + "]"));
200  }
201  }
202  return len_in;
203  }
204

211  // note: maybe it is better to enable Z_SYNC_FLUSH support,
212  // but it will make output result non-deterministic
